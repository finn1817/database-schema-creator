  name string

  sort_order number
  is_active boolean
  created_at timestamp

  id (primary)
  name string
  slug string unique
  sku string unique

  price number
  compare_at_price number

  stock_quantity number

  category_id -> references categories

  weight number
  dimensions text

  created_at timestamp
